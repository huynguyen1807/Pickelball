п»ҝimport { useState, useEffect } from 'react'
import api from '../api/axios'
import styles from '../styles/Dashboard.module.css'

export default function OwnerCourts() {
    const [courts, setCourts] = useState([])
    const [loading, setLoading] = useState(true)
    const [showCreate, setShowCreate] = useState(false)
    const [creating, setCreating] = useState(false)
    const [createForm, setCreateForm] = useState({
        name: '', address: '', description: '', price_per_hour: '', latitude: '', longitude: '', image: ''
    })

    useEffect(() => {
        loadCourts()
    }, [])

    const loadCourts = async () => {
        try {
            const res = await api.get('/courts/my')
            setCourts(res.data)
        } catch (err) {
            console.error('Failed to load courts:', err)
        } finally {
            setLoading(false)
        }
    }

    const handleCreateCourt = async () => {
        if (!createForm.name || !createForm.address || !createForm.price_per_hour) {
            alert('Vui lв”ңв–“ng в”ҖГҰiГҹв•—Гјn tв”ңВ¬n sв”ңГіn, в”ҖГҰГҹв•—ГҜa chГҹв•—Г« vв”ңГЎ giв”ңГӯ')
            return
        }
        setCreating(true)
        try {
            await api.post('/courts', {
                name: createForm.name,
                address: createForm.address,
                description: createForm.description,
                image: createForm.image || null,
                price_per_hour: parseFloat(createForm.price_per_hour),
                latitude: createForm.latitude ? parseFloat(createForm.latitude) : null,
                longitude: createForm.longitude ? parseFloat(createForm.longitude) : null
            })
            setShowCreate(false)
            setCreateForm({ name: '', address: '', description: '', price_per_hour: '', latitude: '', longitude: '', image: '' })
            loadCourts()
        } catch (err) {
            alert(err.response?.data?.message || 'LГҹв•—Г№i tГҹв•‘Гӯo sв”ңГіn')
        } finally {
            setCreating(false)
        }
    }

    const handleToggleActive = async (court) => {
        try {
            await api.put(`/courts/${court.id}`, {
                name: court.name,
                address: court.address,
                description: court.description,
                image: court.image,
                price_per_hour: court.price_per_hour,
                latitude: court.latitude,
                longitude: court.longitude,
                is_active: !court.is_active
            })
            loadCourts()
        } catch (err) {
            alert(err.response?.data?.message || 'LГҹв•—Г№i cГҹв•‘ВЎp nhГҹв•‘ВЎt')
        }
    }

    const handleDelete = async (courtId) => {
        if (!confirm('BГҹв•‘Гӯn chГҹв•‘В»c chГҹв•‘В»n muГҹв•—ГҰn xв”ңв”Ӯa sв”ңГіn nв”ңГЎy?')) return
        try {
            await api.delete(`/courts/${courtId}`)
            loadCourts()
        } catch (err) {
            alert(err.response?.data?.message || 'LГҹв•—Г№i xв”ңв”Ӯa sв”ңГіn')
        }
    }

    const formatPrice = (p) => new Intl.NumberFormat('vi-VN').format(p) + 'в”ҖГҰ'

    if (loading) return <div className={styles.dashboardPage} style={{ textAlign: 'center', padding: '60px 20px' }}>О“Г…в”Ӯ в”ҖГүang tГҹв•‘Гәi...</div>

    return (
        <div className={styles.dashboardPage}>
            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '28px' }}>
                <div>
                    <h1 className="page-title">О“ГңГ–вҲ©в••Г… QuГҹв•‘Гәn lв”ңв•ң sв”ңГіn</h1>
                    <p className="page-subtitle">Thв”ңВ¬m, sГҹв•—ВЎa, xв”ңв”Ӯa sв”ңГіn cГҹв•—Вәa bГҹв•‘Гӯn</p>
                </div>
                <button className="btn btn-primary" onClick={() => setShowCreate(!showCreate)}>+ Thв”ңВ¬m sв”ңГіn mГҹв•—Вўi</button>
            </div>

            {/* Create Court Form */}
            {showCreate && (
                <div className="glass-card" style={{ marginBottom: '20px', animation: 'fadeInUp 0.3s ease' }}>
                    <h3 className={styles.sectionTitle}>вүЎЖ’Г…Ж’вҲ©в••Г… Thв”ңВ¬m sв”ңГіn mГҹв•—Вўi</h3>
                    <div style={{ display: 'flex', flexDirection: 'column', gap: '14px' }}>
                        <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '12px' }}>
                            <div className="input-group">
                                <label>Tв”ңВ¬n sв”ңГіn</label>
                                <input className="input-field" placeholder="VD: Sв”ңГіn Pickleball ABC"
                                    value={createForm.name} onChange={e => setCreateForm(p => ({ ...p, name: e.target.value }))} />
                            </div>
                            <div className="input-group">
                                <label>Giв”ңГӯ / giГҹв•—ВҘ (VNв”ҖГү)</label>
                                <input className="input-field" type="number" placeholder="150000"
                                    value={createForm.price_per_hour} onChange={e => setCreateForm(p => ({ ...p, price_per_hour: e.target.value }))} />
                            </div>
                        </div>
                        <div className="input-group">
                            <label>в”ҖГүГҹв•—ГҜa chГҹв•—Г«</label>
                            <input className="input-field" placeholder="123 в”ҖГүв•һв–‘Гҹв•—ВҘng ABC, QuГҹв•‘ВЎn XYZ, в”ҖГүв”ңГЎ NГҹв•‘в•Ўng"
                                value={createForm.address} onChange={e => setCreateForm(p => ({ ...p, address: e.target.value }))} />
                        </div>
                        <div className="input-group">
                            <label>Mв”ңв”Ө tГҹв•‘Гә</label>
                            <textarea className="input-field" rows={3} placeholder="Mв”ңв”Ө tГҹв•‘Гә sв”ңГіn, tiГҹв•—Г§n в”ңВЎch..." style={{ resize: 'vertical' }}
                                value={createForm.description} onChange={e => setCreateForm(p => ({ ...p, description: e.target.value }))} />
                        </div>
                        <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '12px' }}>
                            <div className="input-group">
                                <label>Vв”ҖвҢҗ в”ҖГҰГҹв•—Г–</label>
                                <input className="input-field" placeholder="16.0544"
                                    value={createForm.latitude} onChange={e => setCreateForm(p => ({ ...p, latitude: e.target.value }))} />
                            </div>
                            <div className="input-group">
                                <label>Kinh в”ҖГҰГҹв•—Г–</label>
                                <input className="input-field" placeholder="108.2022"
                                    value={createForm.longitude} onChange={e => setCreateForm(p => ({ ...p, longitude: e.target.value }))} />
                            </div>
                        </div>
                        <div style={{ display: 'flex', gap: '12px', justifyContent: 'flex-end' }}>
                            <button className="btn btn-secondary" onClick={() => setShowCreate(false)}>HГҹв•—Вәy</button>
                            <button className="btn btn-primary" onClick={handleCreateCourt} disabled={creating}>
                                {creating ? 'О“Г…в”Ӯ...' : 'вүЎЖ’ГҶв•ӣ Lв•һв–‘u sв”ңГіn'}
                            </button>
                        </div>
                    </div>
                </div>
            )}

            {/* Court List */}
            {courts.length > 0 ? courts.map(court => (
                <div key={court.id} className={styles.courtManageCard}>
                    <div className={styles.courtManageIcon}>вүЎЖ’Г…Ж’вҲ©в••Г…</div>
                    <div className={styles.courtManageInfo}>
                        <div className={styles.courtManageName}>{court.name}</div>
                        <div className={styles.courtManageAddress}>{court.address}</div>
                        <div className={styles.courtManageStats}>
                            <span className={styles.courtManageStat}>вүЎЖ’ГҶв–‘ {formatPrice(court.price_per_hour)}/h</span>
                            <span className={styles.courtManageStat}>вүЎЖ’ГҙГҜ {court.booking_count || 0} booking</span>
                            <span className={styles.courtManageStat}>О“ВЎГү {court.avg_rating ? parseFloat(court.avg_rating).toFixed(1) : 'N/A'}</span>
                        </div>
                    </div>
                    <div className={styles.courtManageActions}>
                        <span className={`badge ${court.is_active ? 'badge-green' : 'badge-red'}`}>
                            {court.is_active ? 'HoГҹв•‘Гӯt в”ҖГҰГҹв•—Г–ng' : 'TГҹв•‘Гӯm ngв•һв–‘ng'}
                        </span>
                        <button className="btn btn-secondary btn-sm" onClick={() => handleToggleActive(court)}>
                            {court.is_active ? 'О“Г…в•• TГҹв•‘Гӯm ngв•һв–‘ng' : 'О“Г»в•ў Kв”ңВЎch hoГҹв•‘Гӯt'}
                        </button>
                        <button className="btn btn-danger btn-sm" onClick={() => handleDelete(court.id)}>
                            вүЎЖ’Г№ГҰвҲ©в••Г… Xв”ңв”Ӯa
                        </button>
                    </div>
                </div>
            )) : (
                <div className="glass-card" style={{ textAlign: 'center', padding: '40px 20px', color: 'var(--text-secondary)' }}>
                    вүЎЖ’Г…Ж’вҲ©в••Г… Chв•һв–‘a cв”ңв”Ӯ sв”ңГіn nв”ңГЎo. NhГҹв•‘Г‘n "Thв”ңВ¬m sв”ңГіn mГҹв•—Вўi" в”ҖГҰГҹв•—Гў bГҹв•‘В»t в”ҖГҰГҹв•‘Вәu!
                </div>
            )}
        </div>
    )
}
